export const validateEmail = (email) => {
  const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return regex.test(email);
};

export const displayErrors = (errors) => {
  console.log("ERRORS : ", errors);
  Object.values(errors).forEach((error) => {
    if (error) {
      toast.error(error);
    }
  });
};
